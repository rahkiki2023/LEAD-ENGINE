// ============================================================ // LEAD ENGINE v2 — COMPLETE APPS SCRIPT // Paste this entire file into Extensions → Apps Script // Then set triggers (instructions at the bottom) // ============================================================ // ---- CONFIGURATION (reads from SETTINGS tab) ---- function getSettings() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const s = ss.getSheetByName('SETTINGS').getDataRange().getValues(); return { email: s[0][1], name: s[1][1], phone: s[2][1], }; } // ============================================================ // SCRIPT 1: BUILD DAILY CALL LIST // Runs every morning at 7am // Reads RAW_MLS + INBOUND_LEADS, sorts by priority, // writes fresh list to DAILY_CALL_LIST // ============================================================ function buildCallList() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const mlsSheet = ss.getSheetByName('RAW_MLS'); const inboundSheet = ss.getSheetByName('INBOUND_LEADS'); const outSheet = ss.getSheetByName('DAILY_CALL_LIST'); const mlsData = mlsSheet.getDataRange().getValues(); const inboundData = inboundSheet.getDataRange().getValues(); const priority = ['Appointment', 'Follow Up', 'Contacted', 'Attempted', 'New']; const skip = ['Dead', 'Listing']; let leads = []; // MLS leads — columns: StreetNum(A), StreetName(B), Municipality(C), OrigPrice(D), ListPrice(E), DOM(F), Remarks(G), Status(H), Notes(I), LastContact(J), OwnerName(K), Phone1(L), Phone2(M), Email(N), SkipSource(O), SkipStatus(P), DNC(Q) mlsData.slice(1).forEach(r => { if (!r[0]) return; const status = r[7] || 'New'; const skipStatus = r[15] || 'Not Started'; const dncStatus = r[16] || 'Not Checked'; const phone1 = r[11] || ''; if (skip.includes(status)) return; if (dncStatus === 'On DNC List') return; // never call DNC leads leads.push({ name: `${r[0]} ${r[1]}`, source: r[2], details: `List: ${r[4]} · DOM: ${r[5]}`, notes: r[8] || '', status: status, origin: 'MLS', phone: phone1, dom: parseInt(r[5]) || 0, ownerName: r[10] || '', skipStatus: skipStatus, hasPhone: !!phone1, }); }); // Inbound leads — columns: Date, Name, Phone, Source, Property, Status, Notes, LastContact inboundData.slice(1).forEach(r => { if (!r[1]) return; const status = r[5] || 'New'; if (skip.includes(status)) return; leads.push({ name: r[1], source: r[3], details: r[4], notes: r[6] || '', status: status, origin: 'Inbound', phone: r[2] || '', dom: 0, hasPhone: !!r[2], skipStatus: 'Complete', }); }); // Absentee Owners — columns: OwnerName(A), PropAddr(B), MailAddr(C), City(D), Phone1(E), Phone2(F), Email(G), Source(H), Units(I), EstValue(J), Status(K), Notes(L), LastContact(M), DNC(N) const absenteeSheet = ss.getSheetByName('ABSENTEE_OWNERS'); if (absenteeSheet) { absenteeSheet.getDataRange().getValues().slice(1).forEach(r => { if (!r[0]) return; const status = r[10] || 'New'; const dncStatus = r[13] || 'Not Checked'; const phone = r[4] || ''; if (skip.includes(status)) return; if (dncStatus === 'On DNC List') return; leads.push({ name: r[0], source: r[7] || 'Absentee Owner', details: `${r[1]} · ${r[9] ? 'Est: '+r[9] : ''} · ${r[8] ? r[8]+' units' : ''}`.trim(), notes: r[11] || '', status: status, origin: 'Absentee', phone: phone, dom: 0, hasPhone: !!phone, skipStatus: phone ? 'Complete' : 'Not Started', }); }); } // Landlords — columns: OwnerName(A), PropAddr(B), MailAddr(C), City(D), Phone1(E), Phone2(F), Email(G), Source(H), Units(I), AskingRent(J), Status(K), Notes(L), LastContact(M), DNC(N) const landlordSheet = ss.getSheetByName('LANDLORDS'); if (landlordSheet) { landlordSheet.getDataRange().getValues().slice(1).forEach(r => { if (!r[0]) return; const status = r[10] || 'New'; const dncStatus = r[13] || 'Not Checked'; const phone = r[4] || ''; if (skip.includes(status)) return; if (dncStatus === 'On DNC List') return; leads.push({ name: r[0], source: r[7] || 'Landlord', details: `${r[1]} · ${r[9] ? 'Rent: '+r[9] : ''} · ${r[8] ? r[8]+' units' : ''}`.trim(), notes: r[11] || '', status: status, origin: 'Landlord', phone: phone, dom: 0, hasPhone: !!phone, skipStatus: phone ? 'Complete' : 'Not Started', }); }); } // Sort: skip trace Complete + phone first, then by priority leads.sort((a, b) => { const aReady = a.hasPhone && a.skipStatus === 'Complete' ? 0 : a.hasPhone ? 1 : 2; const bReady = b.hasPhone && b.skipStatus === 'Complete' ? 0 : b.hasPhone ? 1 : 2; if (aReady !== bReady) return aReady - bReady; return priority.indexOf(a.status) - priority.indexOf(b.status); }); // Write to DAILY_CALL_LIST outSheet.clearContents(); outSheet.appendRow(['#', 'Name / Address', 'City / Source', 'Details', 'Notes', 'Status', 'Origin', 'Phone']); // Style header row const headerRange = outSheet.getRange(1, 1, 1, 8); headerRange.setBackground('#1e293b'); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); leads.forEach((l, i) => { const skipBadge = !l.hasPhone ? ' 🔍' : l.skipStatus === 'Complete' ? ' ✅' : ''; outSheet.appendRow([i + 1, l.name + skipBadge, l.source, l.details, l.notes, l.status, l.origin, l.phone, l.ownerName]); // Color code by status const colors = { 'Appointment': '#d1fae5', 'Follow Up': '#fef3c7', 'Contacted': '#ede9fe', 'Attempted': '#fef9c3', 'New': '#f1f5f9', }; outSheet.getRange(i + 2, 1, 1, 8).setBackground(colors[l.status] || '#ffffff'); }); // Auto-resize columns outSheet.autoResizeColumns(1, 8); Logger.log(`Call list built: ${leads.length} leads`); } // ============================================================ // SCRIPT 2: FLAG HIGH-DOM LEADS // Runs every morning at 7am // Highlights RAW_MLS rows by urgency level // ============================================================ function flagHighDOM() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('RAW_MLS'); const data = sheet.getDataRange().getValues(); data.slice(1).forEach((row, i) => { const dom = parseInt(row[5]) || 0; const status = row[7] || ''; const rowNum = i + 2; const range = sheet.getRange(rowNum, 1, 1, 10); if (status === 'Dead' || status === 'Listing') { range.setBackground('#f1f5f9'); // grey out inactive } else if (dom > 60) { range.setBackground('#fee2e2'); // red — very motivated } else if (dom > 30) { range.setBackground('#ffedd5'); // orange — motivated } else if (dom > 14) { range.setBackground('#fef9c3'); // yellow — watch } else { range.setBackground('#ffffff'); // fresh listing } }); Logger.log('DOM flagging complete'); } // ============================================================ // SCRIPT 3: SEND FOLLOW-UP DIGEST EMAIL // Runs every morning at 8am // Emails you a list of all Follow Up leads for the day // ============================================================ function sendFollowUpDigest() { const settings = getSettings(); const ss = SpreadsheetApp.getActiveSpreadsheet(); const mlsData = ss.getSheetByName('RAW_MLS').getDataRange().getValues(); const inboundData = ss.getSheetByName('INBOUND_LEADS').getDataRange().getValues(); let followUps = []; mlsData.slice(1).forEach(r => { if (r[7] === 'Follow Up') { followUps.push({ name: `${r[0]} ${r[1]} (${r[2]})`, notes: r[8] || 'No notes', origin: 'MLS', phone: '', }); } }); inboundData.slice(1).forEach(r => { if (r[5] === 'Follow Up') { followUps.push({ name: `${r[1]}`, notes: r[6] || 'No notes', origin: 'Inbound', phone: r[2] || '', }); } }); if (followUps.length === 0) { Logger.log('No follow-ups today, skipping email'); return; } const today = new Date().toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric' }); let body = `Good morning ${settings.name},\n\n`; body += `You have ${followUps.length} follow-up${followUps.length > 1 ? 's' : ''} scheduled for ${today}:\n\n`; followUps.forEach((f, i) => { body += `${i + 1}. ${f.name} [${f.origin}]\n`; if (f.phone) body += ` 📞 ${f.phone}\n`; body += ` 📝 ${f.notes}\n\n`; }); body += `---\nOpen your Lead Engine to start your calls.\nGood luck today!\n`; GmailApp.sendEmail( settings.email, `☎️ ${followUps.length} Follow-Up${followUps.length > 1 ? 's' : ''} Due Today — ${today}`, body ); Logger.log(`Digest sent: ${followUps.length} follow-ups`); } // ============================================================ // SCRIPT 4: FLAG DUPLICATE LEADS // Trigger: On change (runs when you paste new MLS data) // Highlights duplicate addresses in purple // ============================================================ function flagDuplicates() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('RAW_MLS'); const data = sheet.getDataRange().getValues(); const seen = {}; let dupeCount = 0; data.slice(1).forEach((row, i) => { const addr = `${row[0]} ${row[1]}`.trim().toLowerCase(); if (!addr || addr === ' ') return; const rowNum = i + 2; if (seen[addr] !== undefined) { // Duplicate — highlight purple sheet.getRange(rowNum, 1, 1, 10).setBackground('#ede9fe'); // Add note to first occurrence too sheet.getRange(seen[addr], 1, 1, 10).setBackground('#ede9fe'); dupeCount++; } else { seen[addr] = rowNum; } }); if (dupeCount > 0) { SpreadsheetApp.getUi().alert(`⚠️ ${dupeCount} duplicate address(es) found and highlighted in purple.`); } Logger.log(`Duplicate check: ${dupeCount} dupes found`); } // ============================================================ // WEB APP API — Serves sheet data as JSON to your front-end // Deploy as Web App (Anyone) to connect the Lead Engine UI // ============================================================ function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const tabName = (e && e.parameter && e.parameter.tab) ? e.parameter.tab : 'DAILY_CALL_LIST'; const sheet = ss.getSheetByName(tabName); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({ error: `Tab '${tabName}' not found` })) .setMimeType(ContentService.MimeType.JSON); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1) .filter(row => row.some(cell => cell !== '')) .map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================ // TRIGGER SETUP INSTRUCTIONS // Extensions → Apps Script → Triggers (clock icon) → Add Trigger // // Function | Event Source | Type | Time // --------------------------------------------------------------- // buildCallList | Time-driven | Day timer | 7–8am // flagHighDOM | Time-driven | Day timer | 7–8am // sendFollowUpDigest | Time-driven | Day timer | 8–9am // flagDuplicates | From Sheet | On change | — // ============================================================
